üíª 2. Design Code (full_adder.v)

`timescale 1ns / 1ps module full_adder( input a, input b, input cin, output sum, output cout ); assign sum = a ^ b ^ cin; assign cout = (a & b) | (b & cin) | (a & cin); endmodule

üß™ 3. Testbench Code (tb_full_adder.v)

`timescale 1ns / 1ps module tb_full_adder(); reg a, b, cin; wire sum, cout; full_adder dut (.a(a), .b(b), .cin(cin), .sum(sum), .cout(cout)); initial begin $display("========================================"); $display("FULL ADDER TESTBENCH STARTED"); $display("========================================"); $display("Time\t a\t b\t cin\t sum\t cout"); $display("----------------------------------------"); $monitor("%0t\t %b\t %b\t %b\t %b\t %b", $time, a, b, cin, sum, cout); {a, b, cin} = 3'b000; #10; {a, b, cin} = 3'b001; #10; {a, b, cin} = 3'b010; #10; {a, b, cin} = 3'b011; #10; {a, b, cin} = 3'b100; #10; {a, b, cin} = 3'b101; #10; {a, b, cin} = 3'b110; #10; {a, b, cin} = 3'b111; #10; $display("----------------------------------------"); $display("TEST COMPLETED AT %0t ns", $time); $display("========================================"); $finish; end endmodule
